  audit_injector.py — called by runner before every write
class AuditInjector:
    def append(self, df, pipeline_id, run_id, zone, proc_status='PASSED_DQ'):
        from pyspark.sql import functions as F
        audit_ver = self.get_transform_version(pipeline_id)
        return df \
          .withColumn('SRC_SYSTEM_CD',       F.lit(self.cfg.src_system)) \
          .withColumn('SRC_FILE_NM',         F.lit(self.cfg.source_path)) \
          .withColumn('SRC_ROW_CHECKSUM',    F.md5(F.concat_ws('|', *df.columns))) \
          .withColumn('SRC_EXTRACT_DT',     F.lit(self.extract_ts)) \
          .withColumn('LOAD_DT',            F.current_date()) \
          .withColumn('LOAD_TS',            F.current_timestamp()) \
          .withColumn('LOAD_RUN_ID',        F.lit(run_id)) \
          .withColumn('LOAD_PIPELINE_ID',   F.lit(pipeline_id)) \
          .withColumn('LOAD_BATCH_ID',      F.lit(self.cfg.batch_id)) \
          .withColumn('LOAD_FREQUENCY',     F.lit(self.cfg.frequency)) \
          .withColumn('PROC_STATUS_CD',     F.lit(proc_status)) \
          .withColumn('PROC_ZONE',          F.lit(zone)) \
          .withColumn('PROC_DQ_FLAGS',      F.lit(self.get_dq_flags(run_id))) \
          .withColumn('PROC_TRANSFORM_VER', F.lit(audit_ver)) \
          .withColumn('PROC_REPROCESS_FLG', F.lit('N')) \
          .withColumn('SYS_CREATED_TS',     F.current_timestamp()) \
          .withColumn('SYS_EFF_FROM_DT',    F.current_date()) \
          .withColumn('SYS_EFF_TO_DT',      F.lit('9999-12-31').cast('date')) \
          .withColumn('SYS_CURRENT_FLG',    F.lit('Y')) \
          .withColumn('SYS_SPARK_APP_ID',   F.lit(spark.sparkContext.applicationId)) \
          .withColumn('SYS_ENV_NM',         F.lit(self.get_active_env())) \
          .withColumn('SYS_SCHEMA_VER',     F.lit(self.get_schema_version(pipeline_id)))

class SLAMonitor:
    def check_all_active_pipelines(self):
        for run in self.oracle.get_running_pipelines():
            sla_mins     = run.sla_minutes
            elapsed_mins = (datetime.now() - run.start_ts).total_seconds() / 60
            at_risk_pct  = self.oracle.get_at_risk_pct(run.pipeline_id)  # from SLA_CONFIG
            predicted_completion = self.predict_completion(run)  # based on row velocity

```
        if elapsed_mins >= sla_mins:
            self.fire_breach(run, elapsed_mins, predicted_completion)

        elif elapsed_mins / sla_mins >= (at_risk_pct / 100):
            if not self.already_alerted(run.run_id, 'AT_RISK'):
                self.fire_at_risk(run, elapsed_mins, predicted_completion)

def predict_completion(self, run):
    # Uses rolling avg of last 7 runs for same pipeline
    hist_avg = self.oracle.get_avg_duration(run.pipeline_id, lookback=7)
    rows_done = self.oracle.get_rows_processed(run.run_id)
    rows_total = self.oracle.get_last_total_rows(run.pipeline_id)
    pct_done = rows_done / max(rows_total, 1)
    return run.start_ts + timedelta(seconds=hist_avg / max(pct_done, 0.01))

  dq_engine.py — universal rule executor, no per-rule code
class DQEngine:
    def validate(self, df, rules) -> (passed_df, quarantine_df):
        quarantine_rows = spark.createDataFrame([], df.schema)
        warn_flags      = {}

```
    for rule in rules:
        if   rule.type == 'NOT_NULL':
            failed = df.filter(F.col(rule.column).isNull())
        elif rule.type == 'RANGE':
            failed = df.filter(~F.col(rule.column).between(rule.min_val, rule.max_val))
        elif rule.type == 'REGEX':
            failed = df.filter(~F.col(rule.column).rlike(rule.pattern))
        elif rule.type == 'REFERENTIAL':
            ref_df = spark.table(rule.lookup_table).select(rule.lookup_col)
            failed = df.join(ref_df, df[rule.column] == ref_df[rule.lookup_col], 'left_anti')
        elif rule.type == 'UNIQUENESS':
            failed = df.groupBy(rule.column_list).count().filter(F.col('count') > 1)
        elif rule.type == 'VOLUME_SPIKE':
            last_cnt = self.oracle.get_last_row_count(rule.pipeline_id, rule.lookback_days)
            curr_cnt = df.count()
            if abs(curr_cnt - last_cnt) / max(last_cnt, 1) > rule.threshold / 100:
                warn_flags[rule.rule_id] = 'VOLUME_ANOMALY'
            continue
        elif rule.type == 'CUSTOM_SQL':
            failed = df.filter(expr(f'NOT ({rule.sql_expression})'))

        self.log_dq_result(rule, failed.count(), df.count())
        self.save_samples(failed, rule)

        if   rule.fail_action == 'ABORT':      raise DQAbortException(rule)
        elif rule.fail_action == 'QUARANTINE': quarantine_rows = quarantine_rows.union(failed); df = df.subtract(failed)
        elif rule.fail_action == 'FLAG':       df = df.withColumn('PROC_DQ_FLAGS', ...)

    return df, quarantine_rows
